    code={`
function useNotifyDemo () {
    const { isOpen, onOpen, onClose } = useNotify();
    return (
        <Container type="col" width="100%">
            <Button  color="secondary" onClick={() => onOpen("TOAST")}>
                Toast
            </Button>
            <Toast
                isOpen={isOpen === "TOAST"}
                onClose={onClose}
                delay={3000}
                bottom="2rem"
                left="35%"
                color="secondary"
            >
                A secondary colored toast placed at the bottom of the view port.
            </Toast>
        </Container>
    );
};
    `}
