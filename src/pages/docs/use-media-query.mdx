function useMediaQueryDemo () {
    const [isDesktop] = useMediaQuery(["(min-width: 1024px)"]);
    return (
        <Badge color="success" width={\`\${isDesktop ? '50%' : '100%'}\`}>
            The documentation is rendering in a {isDesktop ? "desktop" : "mobile"}
        </Badge>
    );
};
